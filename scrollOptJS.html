	#**************Load root count and root length data************
	
	#These also expose the raw data globally
	rootCountData <- loadCountData(s_trial, treatmentLevels, treatmentLabels)
	rootLengthData <- loadRootLengthData(s_trial, treatmentLevels, treatmentLabels)
	shootLengthData <- loadRootLengthData(s_trial, treatmentLevels, treatmentLabels, type = "shoots")
	
	
	#**************Basic plots************
	plot_rootCountsQQ(rootCountData)
	plot_rootCounts(rootCountData)
	
	plot_rootLengthsQQ(rootLengthData)
	plot_rootLengths(rootLengthData, TreatmentLabel)
	
	
	plot_shootLengthsQQ(shootLengthData)
	plot_shootLengths(shootLengthData, TreatmentLabel)

	View(rootCountData %>% group_by(Treatment) %>% summarise(mean = mean(RootCount)))

	#TODO note that 95% confidence intervals are not sable unless ~ ESS > 10000 is achieved. So should

	#****Lengths***
	#Generate priors from trial file's specification
	#Priors are based on the estimates of the first bioassay
	rootLPriors <- generate_priors(priors$treatment, priors$meanL)
	
	rootL_Bayes <- analyse_rootLength_Bayes(rootLengthDataRaw, treatmentLevels, iter = 40000, priors = rootLPriors)
	
	#Posterior check
	pp_check(rootL_Bayes, ndraws = 100)
	pp_check(rootL_Bayes, prefix = "ppc", type = "stat_2d", ndraws = 1000)
	pp_check(rootL_Bayes, prefix = "ppc", type = "intervals_grouped", group = "Treatment", ndraws = 10)
	pp_check(rootL_Bayes, type = "stat_grouped", group = "Treatment", stat = "median", ndraws = 1000)
	
	#pp_check(rootL_Bayes, ndraws = 100) + ggtitle("a) Thallus region") + xlab("Root length (log cm)") + ylab("Density")

	rootL_Bayes_PriorCheck <- update(rootL_Bayes, sample_prior = "only")
	pp_check(rootL_Bayes_PriorCheck, ndraws = 1000, type = "stat")+
	  xlim(-30,30)
	
	#Equivalent of R2

	#ROPE limit set as +-0.1 * SDy, measured here as residual standard deviation
	#Table for hypothesis tests
	tbl_rootL <- plot_hypothesisTable(rootL_Bayes, getHypothesisMatrix(), limit = 0.05,#summary(rootL_Bayes)$spec_pars$Estimate * 0.1,
